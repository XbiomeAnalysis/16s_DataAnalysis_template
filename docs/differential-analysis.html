8.3 Dominant taxa

Display the significant taxa with selection using boxplot.

DA_wilcox <- run_wilcox(
   dada2_ps_final,
   group = "Group",
   group_names = c("AA", "BB"),
   transform = "identity",
   norm = "none",
   p_adjust = "BH",
   pvalue_cutoff = 0.05)

plot_topN_boxplot(
    ps = dada2_ps_final,
    da_res = DA_wilcox,
    x_index = "Log2FoldChange (Median)\nAA_vs_BB",
    x_index_cutoff = 0.2,
    y_index = "Pvalue",
    y_index_cutoff = 0.3,
    prevalence = TRUE,
    topN = 3,
    group = "Group")

8.4 Multiple differential analysis by one function

here, we provide the run_multiple_da for obtaining the results list from multiple differential analysis methods.

multiple_res <- run_multiple_da(
       dada2_ps_final,
       group = "Group",
       group_names = c("AA", "BB"),
       da_method = c("aldex", "limma_voom", "mbzinb", "omnibus"),
       norm = "none",
       transform = "identity",
       p_adjust = "none",
       pvalue_cutoff = 0.05)
